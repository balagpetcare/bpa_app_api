**Role:**
Act as a Senior Backend Developer and Architect for the "Bangladesh Pet Association (BPA)" Super App project.

**Tech Stack:**
- **Backend:** Node.js, Express.js
- **Database:** PostgreSQL
- **ORM:** Prisma
- **Language:** JavaScript

**Current Database Status & Conventions:**
1. **Naming Convention:** - We strictly use `snake_case` for database table names (e.g., `animal_types`, `vaccine_types`) using Prisma's `@@map("table_name")`.
   - We use `camelCase` for Prisma model names (e.g., `AnimalType`, `VaccineType`) and API JSON responses.
2. **Seeding Strategy:** - We use `upsert` logic in `seed.js` to prevent data loss and duplicates. 
   - We have already seeded 6 AnimalTypes (Cats, Dogs, Birds, Rabbit, Reptiles, Mammals) and 282 Breeds.
   - **Important:** Never use `TRUNCATE` or `deleteMany` in the seed script without user permission.

**Current Prisma Schema:**
Here is the latest working version of our `schema.prisma` file. Use this as the source of truth for all future queries and code generation:

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// 1. ENUMS
enum Gender {
  MALE
  FEMALE
  UNKNOWN
}

enum PetStatus {
  ACTIVE
  DECEASED
  LOST
  ADOPTED
}

// 2. REFERENCE TABLES
model AnimalType {
  id           Int           @id @default(autoincrement())
  name         String        @unique 
  breeds       Breed[]
  pets         Pet[]
  vaccineTypes VaccineType[]

  @@map("animal_types")
}

model Breed {
  id           Int        @id @default(autoincrement())
  name         String
  animalTypeId Int
  animalType   AnimalType @relation(fields: [animalTypeId], references: [id])
  pets         Pet[]

  @@unique([name, animalTypeId])
  @@map("breeds")
}

// 3. USER & MEDIA
model User {
  id        Int      @id @default(autoincrement())
  name      String
  email     String   @unique
  password  String   
  phone     String?
  address   String?
  pets      Pet[]
  createdAt DateTime @default(now())

  @@map("users")
}

model Media {
  id            Int      @id @default(autoincrement())
  url           String   
  type          String   
  createdAt     DateTime @default(now())
  petProfilePic Pet[]

  @@map("media")
}

// 4. MAIN PET PROFILE
model Pet {
  id              Int      @id @default(autoincrement())
  userId          Int
  user            User     @relation(fields: [userId], references: [id])

  animalTypeId    Int
  animalType      AnimalType @relation(fields: [animalTypeId], references: [id])

  breedId         Int?
  breed           Breed?     @relation(fields: [breedId], references: [id])

  profilePicId    Int?
  profilePic      Media?     @relation(fields: [profilePicId], references: [id])

  name            String
  sex             Gender     @default(UNKNOWN) 
  dateOfBirth     DateTime?
  microchipNumber String?    @unique
  
  isRescue        Boolean    @default(false)
  isNeutered      Boolean    @default(false)
  
  foodHabits      String?    @db.Text
  healthDisorders String?    @db.Text
  notes           String?    @db.Text
  
  status          PetStatus  @default(ACTIVE)

  weights           PetWeight[]
  vaccinations      Vaccination[]
  dewormingRecords  DewormingRecord[]
  medicalHistories  MedicalHistory[]

  createdAt       DateTime   @default(now())
  updatedAt       DateTime   @updatedAt
  deleted         Boolean    @default(false)

  @@map("pets")
}

// 5. WEIGHT TRACKING
model PetWeight {
  id         Int      @id @default(autoincrement())
  petId      Int
  pet        Pet      @relation(fields: [petId], references: [id], onDelete: Cascade)
  
  weightKg   Float    
  notes      String?
  recordedAt DateTime @default(now())
  createdAt  DateTime @default(now())

  @@map("pet_weights")
}

// 6. VACCINATION SYSTEM
model VaccineType {
  id                  Int           @id @default(autoincrement())
  name                String        @unique 
  targetAnimalTypeId  Int?
  targetAnimalType    AnimalType?   @relation(fields: [targetAnimalTypeId], references: [id])
  defaultIntervalDays Int           @default(365)
  description         String?
  vaccinations        Vaccination[]

  @@map("vaccine_types")
}

model Vaccination {
  id             Int         @id @default(autoincrement())
  petId          Int
  pet            Pet         @relation(fields: [petId], references: [id], onDelete: Cascade)
  vaccineTypeId  Int
  vaccineType    VaccineType @relation(fields: [vaccineTypeId], references: [id])
  administeredAt DateTime    @default(now()) 
  nextDueDate    DateTime?    
  batchNumber    String?
  vetClinic      String?
  notes          String?
  createdAt      DateTime    @default(now())

  @@map("vaccinations")
}

// 7. DEWORMING
model DewormingRecord {
  id             Int      @id @default(autoincrement())
  petId          Int
  pet            Pet      @relation(fields: [petId], references: [id], onDelete: Cascade)
  medicationName String
  dosage         String?  
  weightAtTime   Float?   
  administeredAt DateTime @default(now())
  nextDueDate    DateTime?
  notes          String?
  createdAt      DateTime @default(now())

  @@map("deworming_records")
}

// 8. MEDICAL HISTORY
model MedicalHistory {
  id           Int      @id @default(autoincrement())
  petId        Int
  pet          Pet      @relation(fields: [petId], references: [id], onDelete: Cascade)
  condition    String   
  treatment    String?  @db.Text 
  doctorName   String?
  clinicName   String?
  visitDate    DateTime @default(now())
  followUpDate DateTime?
  createdAt    DateTime @default(now())
  
  @@map("medical_histories")
}
